{{- /* opt-image.html — the single reusable image optimizer. Input dict: resource : a Hugo image Resource (page or assets); may be nil alt : alt text class : optional class loading : "lazy" (default) | "eager" maxWidth : optional width cap override (banners pass bannerMaxWidth) src : optional raw src, used only for the passthrough fallback When site.Params.imageOptimize.enabled AND the resource is a processable raster (not svg/gif), emits a WebP primary capped at maxWidth + a responsive srcset + explicit width/height. Resilient: a corrupt/undecodable image (or any process error) falls through to a raw rather than failing the build. */ -}} {{- $r := .resource -}} {{- $o := site.Params.imageOptimize -}} {{- $enabled := and $o $o.enabled -}} {{- $sub := "" -}}{{- with $r }}{{ $sub = .MediaType.SubType }}{{- end -}} {{- $raster := and $r (not (in (slice "svg" "gif") $sub)) -}} {{- $primary := false -}} {{- $srcset := slice -}} {{- if and $enabled $raster -}} {{- $fmt := $o.format | default "webp" -}} {{- $q := int ($o.quality | default 82) -}} {{- $maxW := int (.maxWidth | default $o.maxWidth | default 1600) -}} {{- /* decode the source width defensively — a corrupt image errors here */ -}} {{- $wt := try $r.Width -}} {{- if not $wt.Err -}} {{- $srcW := $wt.Value -}} {{- $pt := cond (gt $srcW $maxW) (try ($r.Resize (printf "%dx %s q%d" $maxW $fmt $q))) (try ($r.Process (printf "%s q%d" $fmt $q))) -}} {{- if not $pt.Err -}} {{- $primary = $pt.Value -}} {{- range $w := (slice 480 960 $maxW) -}} {{- /* clamp to BOTH the source width and the cap (matters when maxWidth < 960) */ -}} {{- if and (le $w $srcW) (le $w $maxW) -}} {{- $dt := try ($r.Resize (printf "%dx %s q%d" $w $fmt $q)) -}} {{- if not $dt.Err -}}{{- $srcset = $srcset | append (printf "%s %dw" $dt.Value.RelPermalink $dt.Value.Width) -}}{{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- if $primary -}} {{- else -}} {{- $src := .src -}}{{- with $r }}{{ $src = .RelPermalink }}{{- end -}} {{- with $src -}} {{- end -}} {{- end -}}
